    PUSH EBX

    POP EAX

; -------------------------------
; validator.asm
; Password comparison module
; -------------------------------

INCLUDE Irvine32.inc

.code

PUBLIC CheckPassword

; -------------------------------------------------------
; CheckPassword
; Compares two strings using REPE CMPSB
;
; Input:
;   ESI = address of user password
;   EDI = address of stored password
;
; Process:
;   Uses REPE CMPSB to compare byte-by-byte
;
; Output:
;   EAX = 1 → match
;   EAX = 0 → mismatch
; -------------------------------------------------------

CheckPassword PROC

    push ecx            ; save ECX (loop counter protection)

    mov ecx,20          ; max password length comparison

    cld                 ; ensure forward direction

    repe cmpsb          ; compare bytes [ESI] vs [EDI]

    jne NotMatch        ; if mismatch found → fail

    mov eax,1           ; passwords match
    jmp Done

NotMatch:
    mov eax,0           ; passwords do not match

Done:
    pop ecx             ; restore ECX
    ret

CheckPassword ENDP

END

    mov ebx,3              ; attempt counter = 3

; LOGIN LOOP (max 3 attempts)
; -------------------------------
TryAgain:

    mov edx,OFFSET msg1
    call WriteString

    ; read user input
    mov edx,OFFSET userPwd
    mov ecx,SIZEOF userPwd
    call ReadString

    ; set pointers for comparison
    mov esi,OFFSET userPwd
    mov edi,OFFSET storedPwd

    cmp eax,1
    je AccessOK

    ; wrong password case
    mov edx,OFFSET denyMsg
    call WriteString
    call Crlf

    dec ebx
    jnz TryAgain

    ; if all attempts fail
    mov edx,OFFSET lockMsg
    call WriteString
    jmp ExitProgram

AccessOK:

    mov edx,OFFSET grantMsg
    call WriteString

ExitProgram:

    exit

main ENDP

; -------------------------------
; String Processing Program
; Uses SCASB, LODSB, MOVSB, STOSB
; -------------------------------

INCLUDE Irvine32.inc

.data

buffer BYTE 100 DUP(0)
buffer2 BYTE 100 DUP(0)

prompt BYTE "Enter String: ",0

upperMsg BYTE "Uppercase letters: ",0
clearMsg BYTE "Buffer cleared.",0

length DWORD ?
upperCount DWORD ?

.code

main PROC

; -------------------------------
; INPUT STRING
; -------------------------------
    mov edx,OFFSET prompt
    call WriteString

    mov edx,OFFSET buffer
    mov ecx,100
    call ReadString

; -------------------------------
; FIND LENGTH USING SCASB
; -------------------------------
    mov edi,OFFSET buffer
    mov ecx,100
    mov al,0               ; search NULL terminator

    cld
    repne scasb            ; scan until AL = 0

    ; compute length = 100 - remaining - 1
    mov eax,100
    sub eax,ecx
    dec eax

    mov length,eax

    mov edx,OFFSET lenMsg
    call WriteString
    mov eax,length
    call WriteInt
    call Crlf

; -------------------------------
; COUNT UPPERCASE USING LODSB
; -------------------------------
    mov esi,OFFSET buffer
    mov ecx,length

    mov upperCount,0

CountLoop:
    lodsb                  ; AL = [ESI], ESI++

    cmp al,'A'
    jb Skip

    cmp al,'Z'
    ja Skip

    inc upperCount

Skip:
    loop CountLoop

    mov edx,OFFSET upperMsg
    call WriteString
    mov eax,upperCount
    call WriteInt
    call Crlf

; -------------------------------
; COPY STRING USING MOVSB
; -------------------------------
    mov esi,OFFSET buffer
    mov edi,OFFSET buffer2
    mov ecx,length
    inc ecx               ; include NULL terminator

    cld
    rep movsb

; -------------------------------
; CLEAR ORIGINAL BUFFER USING STOSB
; -------------------------------
    mov edi,OFFSET buffer
    mov ecx,100
    mov al,0

    cld
    rep stosb

    mov edx,OFFSET clearMsg
    call WriteString
    call Crlf

    exit

main ENDP
END main

highest DWORD 0

.code

; -------------------------------------------------------
; DisplayMatrix PROC
; Displays 4x5 matrix row-wise
;
; Preserves registers using PUSH/POP
; -------------------------------------------------------

DisplayMatrix PROC

    push eax
    push ebx
    push ecx
    push edx
    push esi

    mov esi,OFFSET scores
    mov ecx,4              ; 4 students
    mov ebx,1              ; student number

RowLoop:

    mov edx,OFFSET studentMsg
    call WriteString

    mov eax,ebx
    call WriteInt
    call Crlf

    push ecx               ; save outer loop counter

    mov ecx,5              ; 5 quizzes per student

PrintCols:

    mov eax,[esi]          ; load score
    call WriteInt

    mov al,' '
    call WriteChar
    mov al,' '
    call WriteChar

    add esi,TYPE DWORD

    loop PrintCols

    call Crlf

    pop ecx                ; restore outer loop
    inc ebx

    loop RowLoop

    pop esi
    pop edx
    pop ecx
    pop ebx
    pop eax

    ret

DisplayMatrix ENDP

; -------------------------------
; MAIN PROGRAM
; -------------------------------

main PROC

    call DisplayMatrix

    ; initialize traversal
    mov esi,OFFSET scores
    mov ecx,4
    mov ebx,1

StudentLoop:

    push ecx              ; protect outer loop

    mov eax,0
    mov ecx,5

QuizLoop:

    add eax,[esi]         ; total per student

    ; check highest score
    cmp [esi],highest
    jbe SkipHigh

    mov edx,[esi]
    mov highest,edx

SkipHigh:

    add esi,TYPE DWORD
    loop QuizLoop

    ; print student total
    mov edx,OFFSET studentMsg
    call WriteString

    mov eax,ebx
    call WriteInt

    mov edx,OFFSET totalMsg
    call WriteString
    call WriteInt

    ; compute average
    mov ebx,5
    mov edx,0
    div ebx

    mov edx,OFFSET avgMsg
    call WriteString
    call WriteInt
    call Crlf

    inc ebx
    pop ecx
    loop StudentLoop

    ; print highest score
    mov edx,OFFSET highMsg
    call WriteString
    mov eax,highest
    call WriteInt
    call Crlf

    exit

main ENDP

END main

mov eax,[ebp+8]

mov ebx,9

mul ebx

div ebx


; add 32

; store local variable

; return value already in EAX

ret 4

; =====================================================
; ComputeSum
;
; Input:
; address of array
; number of elements
;
; Uses:
; ESI and ECX preserved
;
; Return:
; EAX = total
;
; =====================================================


ComputeSum PROC USES esi ecx, pArr:PTR DWORD, n:DWORD


LOCAL accumulator:DWORD


mov accumulator,0


mov esi,pArr

mov ecx,n



L1:


mov eax,[esi]


add accumulator,eax


add esi,4


loop L1



mov eax,accumulator


ret

ComputeSum ENDP





; =====================================================
; ComputeAvg
;
; total / n
;
; Return EAX = average
;
; =====================================================


ComputeAvg PROC total:DWORD,n:DWORD



mov eax,total


mov ebx,n


xor edx,edx


div ebx



ret


ComputeAvg ENDP





; =====================================================
; FindHighest
;
; compares array values
;
; Return:
; EAX = highest number
;
; =====================================================

mov esi,pArr


mov ecx,n



; first value as highest

mov eax,[esi]


add esi,4


dec ecx



L2:


cmp [esi],eax


JA updateHigh


jmp next



updateHigh:

mov eax,[esi]



next:

add esi,4


loop L2



ret

PUBLIC CheckPassword


;-------------------------------------
; CheckPassword
;
; Input:
; ESI = user password address
; EDI = stored password address
;
; Output:
; EAX = 1 match
; EAX = 0 wrong
;
; Uses:
; REPE CMPSB
;
;-------------------------------------


CheckPassword PROC


push ecx
push esi
push edi


mov ecx,20        ; compare 20 characters


repe cmpsb        ; compare byte by byte


jne NotMatch



; matched

mov eax,1

jmp Done



NotMatch:

mov eax,0



Done:


pop edi
pop esi
pop ecx


ret


CheckPassword ENDP


END
main.asm
INCLUDE Irvine32.inc


EXTERN CheckPassword:PROC


.data


stored BYTE "12345",0


input BYTE 50 DUP(?)


msg1 BYTE "Enter password: ",0

grant BYTE "Access Granted",0

deny BYTE "Access Denied",0

lockMsg BYTE "System Locked",0


.code


main PROC



mov ecx,3          ; 3 attempts


Again:


mov edx,OFFSET msg1

call WriteString



mov edx,OFFSET input

mov ecx,50

call ReadString



; send addresses

mov esi,OFFSET input

mov edi,OFFSET stored



call CheckPassword



cmp eax,1


je Success



; wrong

mov edx,OFFSET deny

call WriteString

call Crlf



loop Again



; after 3 failures


mov edx,OFFSET lockMsg

call WriteString


jmp ExitProgram




Success:


mov edx,OFFSET grant

call WriteString




ExitProgram:


exit


main ENDP


END main
TASK 2 — String Processing
INCLUDE Irvine32.inc


.data


buffer BYTE 100 DUP(?)

copyBuffer BYTE 100 DUP(?)


msg BYTE "Enter text: ",0

lenMsg BYTE "Length: ",0

upperMsg BYTE "Uppercase letters: ",0


.code


main PROC



; INPUT STRING


mov edx,OFFSET msg

call WriteString


mov edx,OFFSET buffer

mov ecx,100

call ReadString




;----------------------------
; FIND LENGTH USING SCASB
;----------------------------


mov edi,OFFSET buffer


mov ecx,100


mov al,0



repne scasb



; calculate length

mov eax,100

sub eax,ecx

dec eax



mov edx,OFFSET lenMsg

call WriteString


call WriteInt

call Crlf





;----------------------------
; COUNT UPPERCASE
;----------------------------


mov esi,OFFSET buffer


mov ecx,eax      ; length


mov ebx,0        ; counter



CountLoop:


lodsb


cmp al,'A'

jb Skip


cmp al,'Z'

ja Skip



inc ebx



Skip:


loop CountLoop




mov edx,OFFSET upperMsg

call WriteString


mov eax,ebx

call WriteInt

call Crlf




;----------------------------
; COPY STRING
;----------------------------


mov esi,OFFSET buffer

mov edi,OFFSET copyBuffer


mov ecx,100


rep movsb





;----------------------------
; CLEAR ORIGINAL BUFFER
;----------------------------


mov edi,OFFSET buffer


mov al,0

mov ecx,100


rep stosb



mov edx,OFFSET msg

call WriteString


exit


main ENDP


END main

Matrix:

4 students

5 quizzes

INCLUDE Irvine32.inc


.data


grades DWORD 

80,90,70,60,85,
75,88,92,65,80,
90,95,85,100,90,
60,70,75,80,65


highest DWORD 0


rowMsg BYTE "Student:",0

totalMsg BYTE " Total=",0

avgMsg BYTE " Avg=",0



.code



;-----------------------------------
; DisplayMatrix
;
; Uses nested loops
; Preserves registers
;-----------------------------------


DisplayMatrix PROC


push eax
push ebx
push ecx
push edx
push esi



mov esi,OFFSET grades


mov ebx,4        ; rows



RowLoop:


push ebx


mov ecx,5        ; columns


mov eax,0        ; total



ColumnLoop:


mov edx,[esi]


call WriteInt


mov eax,[esi]

add total,eax


add esi,4


loop ColumnLoop



call Crlf



pop ebx


dec ebx


jnz RowLoop





pop esi
pop edx
pop ecx
pop ebx
pop eax


ret


DisplayMatrix ENDP






main PROC



call DisplayMatrix


exit


main ENDP


END main
